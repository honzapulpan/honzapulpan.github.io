rq_diff = [simplify(diff(rq, x)) for x in [a,b]]
sol_full = solvers.solve(rq_diff,(a,b))

sol = []
for sol_a, sol_b in sol_full:
    if (sol_a > 0) & (sol_b > 0): # parametry a,b mus bt vt ne 0
        rq_diff2 = [[simplify(rq.diff(x).diff(y)) for x in [a,b]] for y in [a,b]]
        uxx = simplify(rq_diff2[0][0].subs([(a,sol_a),(b,sol_b)]))
        uxy = simplify(rq_diff2[0][1].subs([(a,sol_a),(b,sol_b)]))
        uyy = simplify(rq_diff2[1][1].subs([(a,sol_a),(b,sol_b)]))
        hess = N(uxx*uyy - uxy**2)
        if hess > 0:
            if N(uxx) > 0:
                sol.append((sol_a,sol_b))
