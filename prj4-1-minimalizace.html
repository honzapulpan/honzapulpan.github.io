Najdeme stacionrn bod a urme, zda se jedn o minimum.

In [59]:
diff_rq = [simplify(diff(rq, x)) for x in [b,c]]
res = solve(diff_rq,(b,c))

diff2_rq = [[simplify(rq.diff(x).diff(y)) for x in [b,c]] for y in [b,c]]
diff2_rq

sol = []
for i in range(0, len(res)):
    uxx = simplify(diff2_rq[0][0].subs([(b,res[i][0]),(c,res[i][1])]))
    uxy = simplify(diff2_rq[0][1].subs([(b,res[i][0]),(c,res[i][1])]))
    uyy = simplify(diff2_rq[1][1].subs([(b,res[i][0]),(c,res[i][1])]))
    hess = N(uxx*uyy - uxy**2)
    if hess > 0:
        if N(uxx) > 0:
            sol.append(res[i])  
            
sol
