# -*- coding: utf-8 -*-
import numpy as np
from scipy import integrate
from IPython.display import display, Markdown, Latex
import matplotlib.pyplot as plt

c = 19999#00; #c ... pocet kroku reseni ODE

def shooting(Eigenvalue, steps):
    def fun(t,y,E):
        return [y[1], -E*(1+t)*y[0]]
    
    sol = integrate.solve_ivp(lambda t, y: fun(t, y, Eigenvalue),
                              [0,1], [0,1],t_eval=np.linspace(0, 1, steps), method='Radau')
    
    return sol.t, sol.y[0,:]

e = 0.000000001       #presnost
a = 5.0         #zacatek intervalu
b = 20.0        #konec intervalu
s = (a + b)/2   #stred intervalu

while (b-a >= e):
    t,u = shooting(s, c)
    value = u[-1]
    if value == 0:
        break
    else:
        t,uA = shooting(a,c)
        t,uS = shooting(s,c)
        if uA[-1] * uS[-1] < 0:
            b = s
        else:
            a = s
        s = (a + b)/2
        
display(Latex('Dominantn vlastn slo $\lambda_1$ : '+ str(s)))

solt, soly = shooting(s,c)
plt.plot(solt, soly, 'r-')
plt.axis([0, 1, 0, 0.5])
plt.show()

from scipy.interpolate import interp1d,splev
from scipy.misc import derivative
import sympy as sp
    
points_range = range(2,41) # rozsah poctu interpolacnich bodu
rp = []

for i in points_range:
    s_r = solt[::int(c/(2*i))]
    s_v = soly[::int(c/(2*i))]
    if s_r[-1] != 1. :
        s_range = np.append(s_r, solt[-1])
        s_value = np.append(s_v, soly[-1])
    else:
        s_range = s_r
        s_value = s_v

    ys_f = interp1d(s_range, s_value)
    xs = np.arange(0,1.0001, 0.0001)
    ys = ys_f(xs)
    
    plt.plot(s_range, s_value, 'o', label='data')
    plt.plot(xs, ys)
    plt.show()
    
    ys_diff_tmp = np.diff(ys)/np.diff(xs)
    ys_diff = np.append(ys_diff_tmp, ys_diff_tmp[-1])
    numerator = np.trapz(ys_diff**2, xs)
    denominator = np.trapz((1+xs)*ys**2, xs)
    
    rp_temp = numerator/denominator
    rp.append(rp_temp)
    display(Latex('Horn odhad dominantnho vlastnho sla $\lambda_1$: ' + str(rp_temp)))
